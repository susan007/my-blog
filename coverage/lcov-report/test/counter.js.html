// counter.js
 
import child from './counter-child'
 
export default {
    template: `
    <div>
      <span class="count">{{ count }}</span>
      <button @click="increment">Increment</button>
      <h3>{{tip}}</h3>
      <button ref="clickMe" @click="clickMe">点我吧</button>
      <child @child="clickChild" :msg="msg"/>
    </div>
  `,
    components: {
        child
    },
    data () {
        return {
            count: 0,
            tip: '说明文档',
            msg: 'father'
        }
    },
    props: {
        age: 10
    },
    methods: {
        increment () {
            this.count++
        },
        clickMe() {
            this.tip = '卡路里'
            alert('click me button')
        },
        clickChild() {
            this.msg = 'child'
        }
    }
}
