### 父传子
* 最常见的一种需求。
* 通过定义props传递参数。
* 单向数据流。

子组件
```html
<template>
  <div>
    <h1>{{title}}</h1>
    <ul>
      <li v-for="item in items" :key="item">{{item}}</li>
    </ul>
  </div>
</template>

<script>
  export default {
    name: 'demo3',
    props: {
      title: {
        type: String,
        default: '',
        required: true
      },
      items: {
        type: Array,
        default: []
      }
    }
  }
</script>
```

父组件
```html
<template>
  <div>
  <!--title和items都是在子组件props中定义好的-->
    <hero :title="heroName" :items="heroSkills" ></hero>
  </div>
</template>

<script>
  import hero from './demo3-son'
  export default {
    name: 'demo3',
    data() {
      return {
        heroName: '盲僧',
        heroSkills: [
          '疾风骤雨',
          '天音波',
          '金钟罩',
          '天雷破',
          '猛龙摆尾'
        ]
      }
    },
    components: {
      hero
    }
  }
</script>
```

### 子传父
* 通过自定义事件实现。
* 父组件注册$on(event)，监听自定义事件。
* 子组件$emit(eventName)触发自定义事件。

子组件注册emit事件
```js
<template>
  <div>
    <h1>{{title}}</h1>
    <ul v-show="show">
      <li v-for="item in items" :key="item">{{item}}</li>
    </ul>
    <button @click="hideDialog">关闭详情</button>
    <button @click="showDialog">查看详情</button>
  </div>
</template>

<script>
  export default {
    name: 'demo3',
    props: {
      title: {
        type: String,
        default: '',
        required: true
      },
      items: {
        type: Array,
        default: []
      },
      show: {
        type: Boolean,
        default: false,
        required: true
      }
    },
    methods: {
      hideDialog() {
          // 注册事件
        this.$emit('hide')
      },
      showDialog() {
        this.$emit('show')
      }
    }
  }
</script>

```

父组件监听详情
```js
<template>
  <div>
  <!--这里的@hide和@show就是监听子组件中注册的事件-->
    <hero :title="heroName" :items="heroSkills" :show="show" @hide="hideDialog" @show="showDialog"></hero>
  </div>
</template>

<script>
  import hero from './demo3-son'
  export default {
    name: 'demo3',
    data() {
      return {
        heroName: '盲僧',
        heroSkills: [
          '疾风骤雨',
          '天音波',
          '金钟罩',
          '天雷破',
          '猛龙摆尾'
        ],
        show: false
      }
    },
    components: {
      hero
    },
    methods: {
      showDialog() {
        this.show = true
      },
      hideDialog() {
        this.show = false
      }
    }
  }
</script>

```

### 兄弟组件通讯
* 使用eventBus
注册一个空白vue作为bus
```js
import Vue from 'vue'
var bus = new Vue()
export default bus
```

组件A中，通过bus注册事件
```js
<template>
  <div>
    我是demo4
    <button @click="showTip">触发事件</button>
  </div>
</template>

<script>
  import bus from './demo6'
  export default {
    name: 'demoFour',
    methods: {
      showTip() {
        // 监听事件
        bus.$emit('hello', 'world')
      }
    }
  }
</script>

```

组件B中通过bus监听事件
```js
<template>
  <div>
    我是demo5
  </div>
</template>

<script>
  import bus from './demo6'
  export default {
    name: 'demoFive',
    created() {
      // 响应事件
      bus.$on('hello', function () {
        alert('hello')
      })
    }
  }
</script>
```

组件C中使用组件A和B
```js
<template>
  <div>
    <demo-four></demo-four>
    <demo-five></demo-five>
  </div>
</template>

<script>
  import demoFour from './demo4'
  import demoFive from './demo5'
  export default {
    name: 'demoSix',
    components: {
      demoFour,
      demoFive
    }
  }
</script>
```
